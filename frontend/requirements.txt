# Oddvark – Python dependencies for the actions server (tools/action-server.py) and Whisper STT.
# Installed by setup.bat (or `pip install -r frontend/requirements.txt`).
# If a package is missing after all, the affected feature degrades cleanly (see GET /capabilities).

psutil            # System info (CPU/RAM/battery/disk), network
pillow            # Screenshots + scaling (vision, see_screen)
mss               # Faster screenshots (alternative to Pillow ImageGrab)
pyautogui         # Mouse/keyboard control, autonomous agent loop (agent_task)
pyperclip         # Read/write the clipboard
pygetwindow       # Window list / window management (optional)

faster-whisper    # Local STT (offline speech recognition, tools/stt-server.py)
                  # The app sends WAV – ffmpeg is NOT required. Model is loaded automatically.

# TTS (XTTS-v2) and Z-Image run in their own venvs (tools/tts-venv, tools/zimage-venv);
# setup.bat creates both – see the README for details.
